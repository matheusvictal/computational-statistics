sample_rej <- function(n){ # fun√ß√£o recebe o tamanho da amostra n
  
  set.seed(2112) #setup da semente

  i <- n_t <- 0 # Inicializa√ß√£o do contador i para o tamanho da amostra e 
  # do contador n_t para o n√∫mero de tentativas 

  a_X <- c() # vetor auxiliar para armazenamento dos valores observados de X

  while(i<n){ # la√ßo para o tamanho da amostra, n√£o √© quebrado at√© o tamanho da
  # amostra ser n
   rej <- TRUE
  
    while(rej){ # la√ßo para a rejei√ß√£o, n√£o √© quebrado at√© que um valor seja  aceito
      n_t <- n_t + 1 
    
      # obten√ß√£o de uma observa√ß√£o da vari√°vel auxiliar pelo m√©todo da invers√£o

  }
